## Reading Colours with `get()` 
The `get()` function allows you to pick up the colour of any pixel on the canvas.
* It returns an array containing the RGBA values: `[red, green, blue, alpha]`.

### Syntax:
* `get(x, y)`: Gets the colour at a specific point.
* `get()`: Captures the entire canvas as an image.

```js
function setup() {
  createCanvas(400, 400);
}

function draw() {
  background(255);
  drawBackground()
  drawMouse()
  
  //Use get() to grab the colour at mouse
 
  push()
    fill(0)

function drawMouse(){
  push()
    stroke(255);
    strokeWeight(4);
    noFill() 
    ellipse(mouseX, mouseY, 30, 30);
  pop()
}


function displayBox(r,g,b){
    fill(0, 150);
  rect(10, 10, 140, 70, 5);
  
  fill(255);
  textSize(16);
  text(`Red: ${r}`, 20, 30);
  text(`Green: ${g}`, 20, 50);
  text(`Blue: ${b}`, 20, 70);
}


function drawBackground(){
    noStroke();

  fill(255, 100, 100)
  rect(0, 0, 200, 200)

  fill(100, 255, 100)
  rect(200, 0, 200, 200)

  fill(100, 100, 255)
  rect(0, 200, 200, 200)

  fill(255, 200, 50)
  rect(200, 200, 200, 200)

}
```
