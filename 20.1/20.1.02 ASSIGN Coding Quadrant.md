function drawBigStar(cx, cy, starSize, col) {
 fill(col)
 stroke(0)
 strokeWeight(1)
 beginShape()
 for (let i = 0; i < 10; i++) {
   let angle = TWO_PI*i/10 - HALF_PI
   let radius = (i % 2 === 0) ? starSize : starSize/2
   let x = cx + cos(angle)*radius
   let y = cy + sin(angle)*radius
   vertex(x, y)
 }
 endShape(CLOSE)
}
